Using a render function to access `isOPen` state from the toggle button

<Playground>
  <Root>
    <div style={{height: 300}}>
      <Popover
        content={({close}) => (
          <div
            style={{
              width: 240,
              height: 240,
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              flexDirection: 'column'
            }}
          >
            <div>Popover Content</div>
          </div>
        )}
      >
        {({isOpen, open}) => (
          <Button onClick={open}>Trigger Popover {isOpen ? '(open)' : '(close)'}</Button>
        )}
      </Popover>
    </div>
  </Root>
</Playground>
